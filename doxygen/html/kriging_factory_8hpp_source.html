29inline std::unique_ptr<KrigingBase> createKriging(const KrigingConfig& cfg)
30{
31 if (cfg.mode == "spatial" || cfg.mode == "space_time") {
32 throw AlgorithmException(

34 "Only temporal Kriging is supported in this release. "

36 }
37 if (cfg.mode != "temporal") {

41 }
42
43 if (cfg.method == "simple") {

45 }
46 if (cfg.method == "ordinary") {
47 return std::make_unique<OrdinaryKriging>(cfg.confidenceLevel);
48 }
49 if (cfg.method == "universal") {

51 throw ConfigException(
52 "createKriging: trendDegree must be in [1, 5], got "
53 + std::to_string(cfg.trendDegree) + ".");
54 }
55 return std::make_unique<UniversalKriging>(cfg.trendDegree,
56 cfg.confidenceLevel);
57 }
58 throw ConfigException(
59 "createKriging: unrecognised method '" + cfg.method
60 + "'. Valid: simple, ordinary, universal.");
